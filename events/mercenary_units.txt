
country_event = {
	id = 48977		

	trigger = {
		war = yes
		money = 1000
		supplies = 1000
		not = { has_country_modifier = Mercenary_Militia }
		not = { has_country_modifier = Mercenary_Infantry }
		not = { has_country_modifier = Mercenary_no }
		not = { has_country_flag = Mercenary_never }
	}
	
	mean_time_to_happen = {
		months = 10
	} 

	title = "Buy a Mercenary Unit"
	desc = "You can choose a Mercenary unit to fight for you. Find it waiting in your production queue"
	picture = "mercenaries"

	option = {
		name = "Mercenary Militia"

		add_country_modifier = {
			name = "Mercenary_Militia"
			duration = 30
		}
		
		load_oob = Mercenary_Militia.txt
		
		ai_chance = { factor = 5 }
		money = -500
		supplies = -500
	}

	option = {
		name = "Mercenary Infantry"

		add_country_modifier = {
			name = "Mercenary_Infantry"
			duration = 60
		}
			
		load_oob = Mercenary_Infantry.txt
		
		ai_chance = { factor = 5 }
		money = -1000
		supplies = -1000
	}
	
	option = {
		name = "Not now"

		add_country_modifier = {
			name = "Mercenary_no"
			duration = 280
		}
		
		ai_chance = { factor = 90 }
	}
	
	option = {
		name = "Never"
		set_country_flag = Mercenary_never
		ai_chance = { factor = 0 }
	}
}
